# Use an official Python runtime as a parent image
FROM python:3.9-slim

# Set the working directory in the container
WORKDIR /app

# Copy the current directory contents into the container at /app
COPY . /app

# Install any needed packages specified in requirements.txt
RUN pip install --no-cache-dir -r requirements.txt

# Expose the ports for the servers
EXPOSE 8001 8002 8003 8004

# Set environment variables
ENV GEMINI_API_KEY=${GEMINI_API_KEY}

# Default command to run multiple servers
CMD ["sh", "-c", "uvicorn 1_chat_server:app --host 0.0.0.0 --port 8001 & \
    uvicorn 2_search_server:app --host 0.0.0.0 --port 8002 & \
    uvicorn 3_thinking_server:app --host 0.0.0.0 --port 8003 & \
    uvicorn 4_rag_server:app --host 0.0.0.0 --port 8004"]
